def generate_reverse_zone(domain,ip_address):
    network = '.'.join(ip_address.split('.')[:3])
    reverse_zone = '.'.join(network.split('.')[::-1]) + ".in-addr.arpa"
    return f""" 
zone "{reverse_zone}" IN {{
    type master;
    file "reverse.{domain}";
    allow-update {{ none; }};
}};
"""
